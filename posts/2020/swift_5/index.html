Swift 拾遗 - 枚举 Enum

2020-02-09 — 5 min read
#Swift 

Basic

// 当枚举的原始值类型是 `Int`、`String`，Swift 会自动分配原始值
enum IntFoo: Int {
    case first
}

enum StrFoo: String {
    case first
}

print(IntFoo.first.rawValue) // 0
print(StrFoo.first.rawValue) // first

// 递归枚举（Recursive enum）：即枚举的关联值类型使用到了该枚举类型本身
indirect enum IndirectBar {
    case first(Int)
    case second(IndirectBar)
}

enum Bar {
    case first(Int)
    indirect case second(Bar)
}

内存布局

关联值

enum Foo {
    case first(Int, Int, Int, Int) // 4 个 Int 值，32 个字节
    case second
}

// 枚举的关联值（Associated Value）占用枚举变量的内存
print(MemoryLayout<Foo>.size)      // 33，实际占用（前 32 位存储关联值，第 33 位存储从 0 开始的次序值）
print(MemoryLayout<Foo>.stride)    // 40，实际分配
print(MemoryLayout<Foo>.alignment) // 8 ，对齐参数

var foo = Foo.first(Int.max, Int.max, Int.max, Int.max)
print(MemoryLayout.size(ofValue: foo)) // 33
withUnsafePointer(to: &foo) { print("\($0)") }
// Memory:
// FF FF FF FF FF FF FF 7F FF FF FF FF FF FF FF 7F
// FF FF FF FF FF FF FF 7F FF FF FF FF FF FF FF 7F
// 00 00 00 00 00 00 00 00
// ⬆️：第一个 case

foo = Foo.second
print(MemoryLayout.size(ofValue: foo)) // 33
// Memory:
// 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00
// 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00
// 01 00 00 00 00 00 00 00
// ⬆️：第二个 case

原始值

enum Bar1: Int {
    case first = 1
    case second
    case third
}

// 枚举的原始值（Raw Value）不占用枚举变量的内存
print(MemoryLayout<Bar1>.size)      // 1（最多约存储 256 个）
print(MemoryLayout<Bar1>.stride)    // 1
print(MemoryLayout<Bar1>.alignment) // 1

enum Bar2 {
    case test
}

// 只有一个 case 且没有关联值时，无需额外存储空间来存储次序值
print(MemoryLayout<Bar2>.size)      // 0
print(MemoryLayout<Bar2>.stride)    // 1
print(MemoryLayout<Bar2>.alignment) // 1

enum Bar3 {
    case test(Int)
}

// 只有一个 case 且有关联值时，也无需额外存储空间来存储次序值
print(MemoryLayout<Bar3>.size)      // 8
print(MemoryLayout<Bar3>.stride)    // 8
print(MemoryLayout<Bar3>.alignment) // 8

enum Bar4 {
    case test0(Int)
    case test1
}

// 有多个 case 且有关联值时，则需要额外的存储空间（不超过 256 个 case 时为额外一个字节来存储次序值）
print(MemoryLayout<Bar4>.size)      // 9（8 + 1）
print(MemoryLayout<Bar4>.stride)    // 16
print(MemoryLayout<Bar4>.alignment) // 8

较为复杂

enum Baz1 {
    case test0
    case test1
    case test2
    case test3
    case test4(Int)
    case test5(Int, Int)
    case test6(Int, Int, Bool, Int) // 8, 8, 8, 8（注意顺序）
}

print(MemoryLayout<Baz1>.size)      // 32
print(MemoryLayout<Baz1>.stride)    // 32
print(MemoryLayout<Baz1>.alignment) // 8

enum Baz2 {
    case test0
    case test1
    case test2
    case test3
    case test4(Int)
    case test5(Int, Int)
    case test6(Int, Bool, Int)
}

print(MemoryLayout<Baz2>.size)      // 25, Why?
print(MemoryLayout<Baz2>.stride)    // 32
print(MemoryLayout<Baz2>.alignment) // 8

enum Baz3 {
    case test0
    case test1
    case test2
    case test3
    case test4(Int)
    case test5(Int, Int)
    case test6(Int, Int, Int, Bool) // 8, 8, 8, 1
}

print(MemoryLayout<Baz3>.size)      // 25
print(MemoryLayout<Baz3>.stride)    // 32
print(MemoryLayout<Baz3>.alignment) // 8

枚举与 switch-case

  • 比较次序值（成员值），将关联值大小的内存绑定到相应的变量
enum Foo {
    case test1(Int, Int, Int)
    case test2(Int, Int)
    case test3(Int)
    case test4(Bool)
    case test5
}

print(MemoryLayout<Foo>.size)      // 25
print(MemoryLayout<Foo>.stride)    // 32
print(MemoryLayout<Foo>.alignment) // 8

let foo = Foo.test1(10, 20, 30) // BREAKPOINT ⚠️
switch foo {
    case let .test1(v1, v2, v3):
        print("test1", v1, v2, v3)
    case let .test2(v1, v2):
        print("test2", v1, v2)
    case let .test3(v1):
        print("test1", v1)
    case let .test4(v1):
        print("test1", v1)
    case .test5:
        print("test5")
    break;
}

Extension

MemoryLayout

MemoryLayout 的 size、stride & alignment：

// size：实际用到的空间大小
print(MemoryLayout<Int>.size)      // 8
// stride：分配占用的空间大小
print(MemoryLayout<Int>.stride)    // 8
// alignment：内存对齐参数（实际分配大小即其倍数）
print(MemoryLayout<Int>.alignment) // 8

var intFoo = 100
print(MemoryLayout.size(ofValue: intFoo)) // 8

withUnsafePointer(to: &intFoo) { print("\($0)") } // 打印值类型变量的内存地址
// Memory:
// 64 00 00 00 00 00 00 00（小端：高字节放在高位）
